Sample Botkit payload

const { Botkit } = require('botkit');
const controller = new Botkit({
  adapterConfig: { clientSigningSecret: process.env.SLACK_SECRET },
  webhook_uri: '/slack/events'
});

controller.on('app_home_opened', async ({ body, context }) => {
  const rooms = await fetchWeekSnapshot(); // reuse Apps Script endpoint

  await controller.client.views.publish({
    token: context.botToken,
    user_id: body.event.user,
    view: {
      type: 'home',
      blocks: [
        {
          type: 'section',
          text: { type: 'mrkdwn', text: '*Room Booking Bot* ¬∑ plan the day' }
        },
        {
          type: 'actions',
          elements: [
            {
              type: 'button',
              text: { type: 'plain_text', text: 'Open booking UI' },
              url: SLACK_CONFIG.bookingAppUrl
            },
            {
              type: 'button',
              text: { type: 'plain_text', text: 'View week grid' },
              action_id: 'open_week_grid'
            }
          ]
        },
        {
          type: 'divider'
        },
        ...rooms.map(room => ({
          type: 'section',
          text: {
            type: 'mrkdwn',
            text: `*${room.label}*  \nNext free slot: ${room.next}`
          }
        }))
      ]
    }
  });
});
